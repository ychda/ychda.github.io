#include <stdio.h>
#include <string.h>

void swap(char *x, char *y) {
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

//int cnt = 0;

void permute(char *a, int index, int len) {
    if (index == len) {
//        cnt++;
        printf("%s\n", a);
    } else {
        for (int i = index; i < len; i++) {
            swap(a + index, a + i);
            permute(a, index + 1, len);
            swap(a + index, a + i); //backtrack
        }
    }
}


int main() {
    char str[] = "123456789";
    int len = strlen(str);
    permute(str, 0, len);
//    printf("%d", cnt);
    return 0;
}
